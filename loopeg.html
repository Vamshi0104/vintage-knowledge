while example

#include<stdio.h>
int main() 
{ 
    
    int j = 1;    /* initialization expression   */
  
    while (j < 6)       /* test expression */
    { 
        printf( "Hello World\n");     
        j++;     /* update expression */
    } 
  
    return 0; 
}

output

Hello World
Hello World
Hello World
Hello World
Hello World




do while example

#include<stdio.h>
int main() 
{ 
    int k = 2;      // Initialization expression 
  
    do
    { 
        // loop body 
        printf( "Hello World\n");     
  
        // update expression 
        k++; 
  
    }  while (k < 1);   // test expression 
  
    return 0; 
}

output

Hello World
