Example 1

//C program for typedef

#include<stdio.h> 
typedef struct Books {
   char title[50];
   char author[50];
   char subject[100];
   int book_id;
} Book;
 
int main( ) {

   Book book;
 
   strcpy( book.title, "C ");
   strcpy( book.author, "V.K"); 
   strcpy( book.subject, "C Tutorial");
   book.book_id = 534;
 
   printf( "Book title : %s\n", book.title);
   printf( "Book author : %s\n", book.author);
   printf( "Book subject : %s\n", book.subject);
   printf( "Book book_id : %d\n", book.book_id);

   return 0;
}

Output

Book  title : C 
Book  author : V.K
Book  subject : C Tutorial
Book  book_id :534
