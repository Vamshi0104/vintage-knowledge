Example 1 (Predefined Macros)

//Predefined Macros

#include<stdio.h>

int main() 
{

   printf("File :%s\n", __FILE__ );
   printf("Date :%s\n", __DATE__ );
   printf("Time :%s\n", __TIME__ );
   printf("Line :%d\n", __LINE__ );
   printf("ANSI :%d\n", __STDC__ );

}

Output

File :test.c
Date :Jun 2 2012
Time :03:36:24
Line :8
ANSI :1




Example 2 (Macro Definition)

// macro definition 

#include<stdio.h>
#define LIMIT 5 

int main() 
{ 
    for (int i = 0; i < LIMIT; i++) 
    { 
        printf("%d \n",i); 
    } 
  
    return 0; 
}

0
1
2
3

Example 3 (Macro with parameter)

// macro with parameter 

#include<stdio.h>
#define AREA(l, b) (l * b) 

int main() 
{ 
    int l1 = 10, l2 = 5, area; 
  
    area = AREA(l1, l2); 
  
    printf("Area of rectangle is: %d", area); 
  
    return 0; 
}

Output

Area of rectangle is: 50




Example 4 (#pragma Directive)

≫ #pragma directive is a special purpose directive and is used to turn on or off some features.
≫ This type of directives are compiler-specific i.e., they vary from compiler to compiler. Some of the #pragma directives are discussed below :
≫ #pragma startup : These directives helps us to specify the functions that are needed to run before the control passes to main()
≫ #pragma exit : These directives helps us to specify the functions that are needed to run just before the control returns from main()

//#pragma Directives

#include<stdio.h>
  
void func1(); 
void func2(); 
  
#pragma startup func1 
#pragma exit func2 
  
void func1() 
{ 
    printf("Inside func1()\n"); 
} 
  
void func2() 
{ 
    printf("Inside func2()\n"); 
} 
  
int main() 
{ 
    void func1(); 
    void func2(); 
    printf("Inside main()\n"); 
  
    return 0; 
}

Output

Inside func1()
Inside main()
Inside func2()

≫ The above code will produce the output as given below when run on GCC compilers.
≫This happens because GCC does not supports #pragma startup or exit.

Output

Inside main()
