auto Example

//C program to illustrate auto
#include<stdio.h> 
int sum(int n1, int n2){
  auto int s;        //declaration of auto(local) variable
  s = n1+n2;
  return s;
}
int main(){
  int i = 2, j = 3, k;
  k = sum(i, j);
  printf("sum is : %d\n", k);
  return 0;
}

Output

sum is : 5




default Example

// C program for  default  

#include<stdio.h>

int Add(int x, int y=20, int z=30)
{
return x + y + z;
}
int main()
{
int rs;
rs = Add(5);
printf("The sum is :%d ",rs);
rs = Add(4,8);
printf("The sum is :%d ",rs);
rs = Add(7,3,4);
printf("The sum is :%d ",rs);
return 0;
}

Output

The sum is : 55
The sum is : 42
