Escape Sequence Example

#include<stdio.h>
int main()
{
char* a = "A\0075"; // \000 is escape sequence wherein each '0' indicate one to three octal (0...7)
printf("%s", a); 
printf("1\a2\a\a3\a");
printf("Vintage\tKnowledge");
printf("hi\nhello");
printf("bye\\welcome");
printf("goodmorning\b\b\bB");
printf("good\rafternoon");
printf("\'coding\n"); 
printf("\"programming"); 
return 0;
}
