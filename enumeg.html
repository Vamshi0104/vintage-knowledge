Example 1

//C program to illustrate enum

#include<stdio.h> 
enum week{mon,tue,wed,thu,fri,sat,sun}; 
  
int main() 
{ 
   int i; 
   for (i=mon; i<=sun; i++)       
      printf("%d ",i); 
        
   return 0; 
}

Output

0 1 2 3 4 5 6




Example 2

// C program to illustrate enum 

#include<stdio.h>

enum State {Working = 1, Failed = 0, Freezed = 0}; 
  
int main() 
{ 
   printf("%d, %d, %d", Working, Failed, Freezed); 
   return 0; 
}

Output

1, 0, 0
